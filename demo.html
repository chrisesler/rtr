An example of routing a static path in an anchor tag:


<html>
<head>
    <meta name="viewport" content="width=device-width, minimum-scale=1.0, initial-scale=1.0, user-scalable=yes">
    <script src="/webcomponentsjs/webcomponents.min.js"></script>
    <link rel="import" href="elements/rtr-router.html">
    <link rel="import" href="demo-staticpath.html">
</head>
<body unresolved>
    <rtr-router>
        <!-- route paths must start with a "/" character -->
        <demo-staticpath method="get" path="/demo-staticpath" handler="get"></demo-staticpath>
    </rtr-router>
</body>
</html>
